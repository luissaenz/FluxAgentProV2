## 5. Plan de Implementación: Run Transcript

### Paso 1: Backend - Endpoint GET /transcripts/{task_id}

```python
# src/api/routes/transcripts.py (NUEVO)
from fastapi import APIRouter, Depends
from ..middleware import require_org_id
from ...db.session import get_tenant_client

router = APIRouter(prefix="/transcripts", tags=["transcripts"])

@router.get("/{task_id}")
async def get_transcript(
    task_id: str,
    org_id: str = Depends(require_org_id),
):
    """Obtiene eventos de dominio para un task_id específico."""
    with get_tenant_client(org_id) as db:
        # Verificar task existe
        task = db.table("tasks").select("id, flow_type, status").eq("id", task_id).maybe_single().execute()
        if not task.data:
            from fastapi import HTTPException
            raise HTTPException(status_code=404, detail="Task not found")
        
        # Obtener eventos
        events = (
            db.table("domain_events")
            .select("id, event_type, payload, sequence, created_at")
            .eq("aggregate_id", task_id)
            .order("sequence")
            .execute()
        )
    
    return {
        "task_id": task_id,
        "flow_type": task.data.get("flow_type"),
        "status": task.data.get("status"),
        "events": events.data or []
    }
```

### Paso 2: Frontend - Hook useFlowTranscript.ts

```typescript
// dashboard/hooks/useFlowTranscript.ts
import { useQuery, useEffect, useState } from '@tanstack/react-query'
import { api } from '@/lib/api'
import { createClient } from '@/lib/supabase/client'

export function useFlowTranscript(orgId: string, taskId: string) {
  const [liveEvents, setLiveEvents] = useState<any[]>([])
  const [isLive, setIsLive] = useState(false)

  // Histórico
  const { data, isLoading } = useQuery({
    queryKey: ['transcript', taskId],
    queryFn: () => api.get(`/transcripts/${taskId}`),
    enabled: !!orgId && !!taskId,
    refetchInterval: 5000, // Polling cada 5s
  })

  // Realtime subscription (opcional)
  useEffect(() => {
    if (!orgId || !taskId) return
    const supabase = createClient()
    const channel = supabase.channel(`transcript:${taskId}`)
    
    channel.on('postgres_changes', {
      event: 'INSERT',
      schema: 'public',
      table: 'domain_events',
      filter: `aggregate_id=eq.${taskId}`,
    }, (payload) => {
      setLiveEvents(prev => [...prev, payload.new])
      setIsLive(true)
    }).subscribe()

    return () => { supabase.removeChannel(channel); setIsLive(false) }
  }, [orgId, taskId])

  // Combinar histórico + live
  const allEvents = [...(data?.events ?? []), ...liveEvents]
  
  return {
    events: allEvents,
    flowType: data?.flow_type,
    status: data?.status,
    isLoading,
    isLive,
  }
}
```
